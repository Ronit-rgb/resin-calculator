<body>
  <h2>Resin Dilution Calculator</h2>
  <label>Initial Resin %</label>
  <input id="r1" type="number" step="any" />

  <label>Final Resin %</label>
  <input id="r2" type="number" step="any" />

  <label>Initial Weight (kg)</label>
  <input id="w1" type="number" step="any" />

  <button onclick="calculate()">Calculate</button>
  <p id="result"></p>

  <script>
    function calculate() {
      let r1 = parseFloat(document.getElementById("r1").value);
      let r2 = parseFloat(document.getElementById("r2").value);
      let w1 = parseFloat(document.getElementById("w1").value);

      if (!r1 || !r2 || !w1 || r2 === 0) {
        document.getElementById("result").innerText = "Please enter valid numbers.";
        return;
      }

      let w2 = (r1 * w1) / r2;
      let solventToAdd = w2 - w1;

      document.getElementById("result").innerText =
        `You need to add ${solventToAdd.toFixed(2)} kg of solvent.`;
    }
  </script>
